import os
import requests
import time

# 带有重试机制的 FLUX.1 生成代码
def generate_image_with_retry(prompt, output_path, retries=3, delay=5):
    url = "https://api.pollinations.ai/p/" + requests.utils.quote(prompt)
    
    for attempt in range(retries):
        try:
            print(f"Sending request to Pollinations (Attempt {attempt+1}/{retries})...")
            response = requests.get(url, timeout=30)
            
            if response.status_code == 200:
                with open(output_path, 'wb') as f:
                    f.write(response.content)
                return True
            else:
                print(f"Error: Status code {response.status_code}")
        except Exception as e:
            print(f"Exception during image generation: {str(e)}")
            
        time.sleep(delay)
    return False
